    ```elixir
    @doc """
    Injects a broadcast message into a paused agent's message history.
    If the agent is not paused (no paused_state), falls back to send_message/2.
    """
    def inject_broadcast(pid, text) when is_pid(pid) do
    ```

    And the handler:
    ```elixir
    def handle_call({:inject_broadcast, text}, _from, %{status: :paused, paused_state: ps} = state) when ps != nil do
      user_message = %{role: :user, content: text}
      updated_ps = %{ps | messages: ps.messages ++ [user_message]}
      {:reply, :ok, %{state | paused_state: updated_ps}}
    end

    # For non-paused agents, delegate to send_message
    def handle_call({:inject_broadcast, text}, from, state) do
      handle_call({:send_message, text}, from, state)
    end
    ```

    5. Add a 4th branch to `handle_event("send_message", ...)` for broadcast. The case statement currently matches on `socket.assigns.reply_target`. Change the `nil ->` branch to check broadcast_mode:

    ```elixir
      if socket.assigns.broadcast_mode do
        # Broadcast to all team agents
        team_id = socket.assigns.team_id
        agents = Loomkin.Teams.Manager.list_agents(team_id)

        Enum.each(agents, fn agent ->
          Task.Supervisor.start_child(Loomkin.Teams.TaskSupervisor, fn ->
            try do
              broadcast_text = "[Broadcast from Human]: #{trimmed}"
              # Use inject_broadcast for paused agents, send_message for active ones
              case Loomkin.Teams.Agent.get_status(agent.pid) do
                :paused -> Loomkin.Teams.Agent.inject_broadcast(agent.pid, broadcast_text)
                _ -> Loomkin.Teams.Agent.send_message(agent.pid, broadcast_text)
              end
            catch
              :exit, _ -> :ok
            end
          end)
        end)

        broadcast_event = %{
          id: Ecto.UUID.generate(),
          type: :human_broadcast,
          agent: "You",
          content: trimmed,
          timestamp: DateTime.utc_now(),
          expanded: false,
          metadata: %{from: "You", to: "All Agents", agent_count: length(agents)}
        }

        {:noreply,
         socket
         |> push_activity_event(broadcast_event)
         |> assign(input_text: "", last_user_message: %{text: trimmed, to: "All Agents"})
         |> push_event("clear-input", %{})}
      else
        # Existing Architect pipeline branch (move existing nil branch code here)
        ...
      end
    ```
